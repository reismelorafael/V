const SEV_ORDER = ["CRÍTICO", "ALTO", "MÉDIO", "BAIXO"];
const SEV_COLOR = { CRÍTICO: "#ff3b3b", ALTO: "#ff8c00", MÉDIO: "#f0c040", BAIXO: "#4fc3f7" };

function count(sev) { return files.filter(f => f.sev === sev).length; }

export default function App() {
  const [filter, setFilter] = useState("TODOS");
  const [open, setOpen] = useState(null);

  const shown = filter === "TODOS" ? files : files.filter(f => f.sev === filter);

  return (
    <div style={{ fontFamily: "'Courier New', monospace", background: "#080c12", minHeight: "100vh", color: "#b8ccd8" }}>
      {/* header */}
      <div style={{ background: "#090e18", borderBottom: "1px solid #0e2030", padding: "20px 28px" }}>
        <div style={{ fontSize: 11, color: "#2a5a7a", letterSpacing: 4, marginBottom: 4 }}>RAFAELIA · VECTRAS-VM-ANDROID · AUDITORIA</div>
        <h1 style={{ margin: 0, fontSize: 18, color: "#d0e8ff", fontWeight: 700 }}>
          Arquivos com Erros — {files.length} ficheiros · {files.reduce((a,f)=>a+f.errors.length,0)} problemas
        </h1>
        <div style={{ display: "flex", gap: 8, marginTop: 12, flexWrap: "wrap" }}>
          {["TODOS", ...SEV_ORDER].map(s => (
            <button key={s} onClick={() => { setFilter(s); setOpen(null); }} style={{
              background: filter === s ? "#0d1e30" : "transparent",
              border: `1px solid ${filter === s ? "#1a4060" : "#0e1e2c"}`,
              color: SEV_COLOR[s] || "#6a9ab8",
              padding: "4px 14px", cursor: "pointer", fontSize: 11, letterSpacing: 1,
            }}>
              {s}{s !== "TODOS" ? ` (${count(s)})` : ` (${files.length})`}
            </button>
          ))}
        </div>
      </div>

      <div style={{ padding: "20px 28px" }}>
        {shown.map((f, fi) => {
          const isOpen = open === fi;
          const col = SEV_COLOR[f.sev];
          return (
            <div key={fi} style={{ marginBottom: 6 }}>
              <div
                onClick={() => setOpen(isOpen ? null : fi)}
                style={{
                  background: "#090e18", border: `1px solid ${col}28`,
                  padding: "11px 16px", cursor: "pointer",
                  display: "flex", alignItems: "center", gap: 12
                }}
              >
                <span style={{
                  fontSize: 9, fontWeight: 700, color: col, background: `${col}16`,
                  border: `1px solid ${col}44`, padding: "2px 7px", letterSpacing: 1,
                  minWidth: 54, textAlign: "center"
                }}>{f.sev}</span>
                <span style={{ flex: 1, fontSize: 12, color: "#90b8d0", fontFamily: "monospace" }}>
                  {f.path}
                </span>
                <span style={{ fontSize: 10, color: "#1a4060" }}>{f.errors.length} erros</span>
                <span style={{ color: "#1a4060", fontSize: 11 }}>{isOpen ? "▲" : "▼"}</span>
              </div>

              {isOpen && (
                <div style={{ background: "#05080f", border: `1px solid ${col}18`, borderTop: "none", padding: "10px 16px 14px" }}>
                  {f.errors.map((e, ei) => (
                    <div key={ei} style={{ display: "flex", gap: 10, padding: "8px 0", borderBottom: ei < f.errors.length - 1 ? "1px solid #0a1520" : "none" }}>
                      <span style={{
                        fontSize: 9, color: col, border: `1px solid ${col}44`,
                        padding: "1px 6px", height: "fit-content", whiteSpace: "nowrap",
                        marginTop: 2, minWidth: 90, textAlign: "center"
                      }}>{e.tag}</span>
                      <span style={{ fontSize: 12, color: "#7a9ab8", lineHeight: 1.7 }}>{e.text}</span>
                    </div>
                  ))}
                </div>
              )}
            </div>
          );
        })}
      </div>

      <div style={{ padding: "0 28px 28px", maxWidth: 800 }}>
        <div style={{ background: "#090e18", border: "1px solid #0e2030", padding: 14, fontSize: 11, color: "#3a6a8a", lineHeight: 2 }}>
          <span style={{ color: "#4fc3f7" }}>F_gap síntese: </span>
          <span style={{ color: "#ff3b3b" }}>zero_compat.h ausente</span> é o root cause de 60% dos erros —
          sem ele cada arquivo resolve stdlib por conta própria criando divergência.
          <span style={{ color: "#ff8c00" }}> JNI_BUILD=ON default</span> mascara todos os erros baremetal em build normal.
          Fix mínimo: criar zero.h + zero_compat.h + setar RMR_JNI_BUILD=OFF em target baremetal.
        </div>
      </div>
    </div>
  );
}
